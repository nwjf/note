vue-router
	SPA: Single Page Application 单页面应用



1.index.js路由文件	// 路由模块

		// 引入vue
		import Vue from 'vue'
		import VueRouter from 'vue-router'

		// use
		Vue.use(VueRouter)
		// 定义路由
	var router = new VueRouter({
		routes : [
			{
				path : '/lol',
				name : 'lol',
				component : Lol,
				children : [
					{
						自路由
					}

				]
			}
		]
	})


	// 暴露模块
	export default router
	export default $http

2.main.js文件
	import router form './router/index.js'
	new Vue({
	  el: '#app',
	  router,
	  render: h => h(App)
	})
